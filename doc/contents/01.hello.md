```js
//모듈선언
angular.module('app',[]);

//컨트롤러 선언 및 구현
angular.module('app')
.controller('controller',['$scope',function($scope){
    $scope.title = '푸통푸통 앵귤러';
}]);

//앵귤러 실행
angular.element(document).ready(function(){
    angular.bootstrap(document,['app']);
});
```

```html
<div ng-controller="controller">
    <h1 ng-bind="title"></h1>
    <div>
        <p>안녕 {{name}}</p>
        <p><input type="text" ng-model="name"></p>
    </div>
</div>
```
