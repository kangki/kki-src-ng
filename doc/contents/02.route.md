```js
// 모듈 선언
var app = angular.module('app', ['ngRoute']);

// Route 설정
app.config(['$routeProvider',function($routeProvider){
    $routeProvider.when('/',{
        controller:'viewController',
        templateUrl:'view.html'
    });
    $routeProvider.otherwise({redirectTo:'/'});
}]);
```
